        return max(min_allowed, min(max_allowed, adaptive_value))

    def calculate_all_thresholds(

                    sensor, stats, standard_config[key], level

class OnlineStatistics:
    """ì˜¨ë¼ì¸ ì¦ë¶„ í†µê³„ ê³„ì‚°ê¸° (Welford's Algorithm)"""

    def __init__(self):
        self.n = 0
        self.mean = 0.0
        self.M2 = 0.0  # ë¶„ì‚° ê³„ì‚°ìš©
        self.min_val = float('inf')
        self.max_val = float('-inf')

        # ë°±ë¶„ìœ„ìˆ˜ìš© Reservoir Sampling
        self.reservoir_size = 10000
        self.reservoir = []

    def update(self, value: float):
        """ìƒˆ ê°’ìœ¼ë¡œ í†µê³„ ì—…ë°ì´íŠ¸ (O(1) ë³µì¡ë„)"""
        self.n += 1
        delta = value - self.mean
        self.mean += delta / self.n
        delta2 = value - self.mean
        self.M2 += delta * delta2

        self.min_val = min(self.min_val, value)
        self.max_val = max(self.max_val, value)

        # Reservoir sampling for percentiles
        if len(self.reservoir) < self.reservoir_size:
            self.reservoir.append(value)
        else:
            # í™•ë¥ ì  êµì²´
            import random
            idx = random.randint(0, self.n - 1)
            if idx < self.reservoir_size:
                self.reservoir[idx] = value

    @property
    def variance(self) -> float:
        return self.M2 / self.n if self.n > 1 else 0.0

    @property
    def std(self) -> float:
        return self.variance ** 0.5

    def get_percentile(self, p: float) -> float:
        """ë°±ë¶„ìœ„ìˆ˜ ê³„ì‚° (ê·¼ì‚¬ê°’)"""
        if not self.reservoir:
            return 0.0
        sorted_reservoir = sorted(self.reservoir)
        idx = int(len(sorted_reservoir) * p / 100)
        return sorted_reservoir[min(idx, len(sorted_reservoir) - 1)]
